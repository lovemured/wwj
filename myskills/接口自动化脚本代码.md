# 接口自动化脚本代码 Skill（Python + pytest + requests）
> 目标：提供一套可复用的**接口自动化脚本工程骨架 + 关键代码模板**，适用于 REST/HTTP 接口自动化测试。支持多环境配置、密钥管理、鉴权与签名、统一断言、日志与可复现 curl（不强依赖 Swagger，curl 也可作为输入/复现方式）。

## 0. 这份 Skill 解决什么

用统一方式沉淀接口自动化脚本，保证：
- 可扩展：新增接口/用例不乱长
- 可维护：分层清晰（配置 / 客户端 / 鉴权 / 用例 / 数据）
- 可定位：日志 + 可复现请求（生成 curl）
- 可迁移：多环境配置隔离，密钥不入库
- 可增强：支持签名、Token 自动刷新、重试/超时、Schema 校验（可选）

## 3. 配置与密钥管理（config.py）

### 3.1 原则
1) 仓库只提交 `.env.example`，真实密钥用环境变量注入  
2) 环境切换使用 `APP_ENV=sit|uat|prod`，不要靠改代码  
3) 日志对 token/手机号/身份证等做脱敏（建议）  

### 3.2 示例代码

```python
# src/app/config.py
from pydantic_settings import BaseSettings, SettingsConfigDict
from pydantic import Field

class Settings(BaseSettings):
    model_config = SettingsConfigDict(env_file=".env", env_file_encoding="utf-8")

    app_env: str = Field(default="sit", alias="APP_ENV")
    base_url: str = Field(alias="BASE_URL")
    timeout_sec: float = Field(default=10, alias="TIMEOUT_SEC")

    # 鉴权（示例：账号密码换 token）
    username: str | None = Field(default=None, alias="API_USERNAME")
    password: str | None = Field(default=None, alias="API_PASSWORD")

    # 签名（示例：HMAC）
    sign_secret: str | None = Field(default=None, alias="SIGN_SECRET")

def load_settings() -> Settings:
    return Settings()
```

`.env.example`（示例）：
```env

BASE_URL=https://sit.example.com
TIMEOUT_SEC=10

API_USERNAME=test_user
API_PASSWORD=please_set_me

SIGN_SECRET=please_set_me
```

## 4. HTTP 客户端封装（http_client.py）

### 4.1 必须做的事情
- 统一 base_url、timeout
- 统一请求入口（避免 tests 里随处 `requests.get/post`）
- 失败时输出可复现 curl
- 统一错误处理（网络错误/超时/非 JSON）变成可读异常

### 4.2 示例代码

```python
# src/app/http_client.py
import json
import time
import requests
from typing import Any

class HttpError(Exception):
    pass

def _to_curl(method: str, url: str, headers: dict[str, str], params: dict[str, Any] | None, json_body: Any | None) -> str:
    parts = [f"curl -X {method.upper()} '{url}'"]
    for k, v in (headers or {}).items():
        if v is None:
            continue
        parts.append(f"-H '{k}: {v}'")
    if json_body is not None:
        body = json.dumps(json_body, ensure_ascii=False)
        parts.append(f"--data '{body}'")
    return " \\\n+  ".join(parts)

class ApiClient:
    def __init__(self, base_url: str, timeout_sec: float = 10):
        self.base_url = base_url.rstrip("/")
        self.timeout_sec = timeout_sec
        self.session = requests.Session()
        self.default_headers: dict[str, str] = {
            "Content-Type": "application/json",
            "Accept": "application/json",
        }

    def set_header(self, k: str, v: str):
        self.default_headers[k] = v

    def request(
        self,
        method: str,
        path: str,
        *,
        headers: dict[str, str] | None = None,
        params: dict[str, Any] | None = None,
        json_body: Any | None = None,
    ) -> requests.Response:
        url = self.base_url + path
        merged_headers = {**self.default_headers, **(headers or {})}

        start = time.time()
        try:
            resp = self.session.request(
                method=method,
                url=url,
                headers=merged_headers,
                params=params,
                json=json_body,
                timeout=self.timeout_sec,
            )
        except requests.RequestException as e:
            curl = _to_curl(method, url, merged_headers, params, json_body)
            raise HttpError(f"请求失败：{e}\n复现命令：\n{curl}") from e
        finally:
            cost = (time.time() - start) * 1000

        # 失败时携带 curl，便于定位（按团队需要扩展：>=400 都打印等）
        if resp.status_code >= 500:
            curl = _to_curl(method, url, merged_headers, params, json_body)
            raise HttpError(
                f"服务端错误 {resp.status_code}，耗时 {cost:.1f}ms\n"
                f"复现命令：\n{curl}\n"
                f"响应片段：{resp.text[:500]}"
            )

        return resp

    def get(self, path: str, **kwargs):
        return self.request("GET", path, **kwargs)

    def post(self, path: str, **kwargs):
        return self.request("POST", path, **kwargs)
```

## 5. Token 自动管理与 401 自动刷新（auth.py）

### 5.1 设计要点
- TokenManager 缓存 token
- 每次请求前注入 Authorization
- 收到 401 自动刷新一次再重试（避免死循环）

### 5.2 示例代码

```python
# src/app/auth.py
from dataclasses import dataclass
from typing import Any
from .http_client import ApiClient, HttpError

@dataclass
class TokenManager:
    client: ApiClient
    username: str
    password: str
    token: str | None = None

    def login(self) -> str:
        resp = self.client.post("/api/v1/login", json_body={"username": self.username, "password": self.password})
        if resp.status_code != 200:
            raise HttpError(f"登录失败：{resp.status_code} {resp.text}")
        data: dict[str, Any] = resp.json()
        self.token = data["data"]["token"]
        return self.token

    def get_token(self) -> str:
        if not self.token:
            return self.login()
        return self.token

def auth_request(client: ApiClient, token_mgr: TokenManager, method: str, path: str, **kwargs):
    token = token_mgr.get_token()
    headers = kwargs.pop("headers", {}) or {}
    headers["Authorization"] = f"Bearer {token}"
    resp = client.request(method, path, headers=headers, **kwargs)

    if resp.status_code == 401:
        token_mgr.login()
        headers["Authorization"] = f"Bearer {token_mgr.get_token()}"
        resp = client.request(method, path, headers=headers, **kwargs)
    return resp
```

## 6. 签名（HMAC 示例，按你们规则替换）（sign.py）

适用：AK/SK、请求签名、防重放。

```python
# src/app/sign.py
import hmac
import hashlib
import time
import json
from typing import Any

def build_signature(secret: str, method: str, path: str, body: Any | None) -> tuple[str, str]:
    ts = str(int(time.time()))
    payload = f"{ts}\n{method.upper()}\n{path}\n"
    if body is not None:
        payload += json.dumps(body, separators=(",", ":"), ensure_ascii=False)
    sig = hmac.new(secret.encode("utf-8"), payload.encode("utf-8"), hashlib.sha256).hexdigest()
    return ts, sig
```

请求时注入（示例）：
```python
from src.app.sign import build_signature

ts, sig = build_signature(settings.sign_secret, "POST", "/api/v1/orders", body)
headers = {"X-Timestamp": ts, "X-Signature": sig}
resp = client.post("/api/v1/orders", headers=headers, json_body=body)
```

## 7. 通用断言（assertions.py）

```python
# src/app/assertions.py
from typing import Any
import jsonschema

def assert_http_ok(resp, expected_status: int = 200):
    assert resp.status_code == expected_status, f"HTTP={resp.status_code}, body={resp.text}"

def assert_biz_code(data: dict[str, Any], expected_code: int = 0):
    actual = data.get("code")
    assert actual == expected_code, f"biz_code={actual}, msg={data.get('msg')}, data={data}"

def assert_schema(data: Any, schema: dict[str, Any]):
    jsonschema.validate(instance=data, schema=schema)
```

```python
import pytest
from src.app.config import load_settings
from src.app.http_client import ApiClient
from src.app.auth import TokenManager

@pytest.fixture(scope="session")

    return load_settings()

    return ApiClient(base_url=settings.base_url, timeout_sec=settings.timeout_sec)

@pytest.fixture(scope="session")
def token_mgr(client, settings):
    if not settings.username or not settings.password:
        pytest.skip("未配置 API_USERNAME/API_PASSWORD，跳过需要登录的用例")
    return TokenManager(client=client, username=settings.username, password=settings.password)
```

用例 `tests/test_order_create.py`
```python
import yaml
import pytest
from src.app.auth import auth_request
from src.app.assertions import assert_http_ok, assert_biz_code

def load_cases():
    with open("tests/data/order_create.yaml", "r", encoding="utf-8") as f:
        return yaml.safe_load(f)

@pytest.mark.parametrize("case", load_cases(), ids=lambda c: c["name"])
def test_create_order(case, client, token_mgr):
    resp = auth_request(client, token_mgr, "POST", "/api/v1/orders", json_body=case["body"])
    assert_http_ok(resp, expected_status=case["expected"]["http_status"])

    # 如果 4xx 不一定有统一返回结构，这里按你们规范调整
    data = resp.json()
    assert_biz_code(data, expected_code=case["expected"]["biz_code"])
```

## 10. “curl 也行”：把 curl 变成资产的最佳实践

### 10.1 推荐约定
- 每条用例都必须有“可复现 curl”（失败也要能复现）
- 用例失败时：日志输出 curl + requestId/traceId（如有）

### 10.2 最低成本落地
上面的 `ApiClient` 已在异常时附带 curl；你也可以在断言失败时打印请求摘要：
```python
req = resp.request
print("复现：", req.method, req.url, req.headers, req.body)
```

## 11. 强约束工程规范（建议写进团队 Code Review Checklist）

- 不在 tests 里硬编码 base_url（统一 Settings）
- 不在 tests 里散落 token/sign（统一 auth/sign）
- 所有请求必须有 timeout；重试谨慎（幂等才可重试）
- 日志/报告不得输出明文密码与完整 token（脱敏）
- 每个 P0 接口至少覆盖：正向 + 参数校验 + 认证/权限（按风险扩展安全/性能）
